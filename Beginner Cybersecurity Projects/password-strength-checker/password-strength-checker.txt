pwd = input("Enter a password: ")

#list to store problems we find

issues = []

#check if password is too short

if len(pwd) < 8:
issues.append("Too short")

#check if it has at least one upper case letter
 
if not any(c.isupper() for c in pwd):
issues.append("No upper case letter")

#check if it has at least one lower case letter

#
if not any(c.islower() for c in pwd):
issues.append("No lower case letter")  

#check if it has at least one number

if not any(c.isdigit() for c in pwd):
issues.append("No number")

 #show results

if len(issues) == 0:
print("Password looks ok")
else:
print("Weak password")
for item in issues:
print("-", item)
