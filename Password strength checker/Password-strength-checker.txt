# Ask the user for a password
password = input("Enter a password to check: ")


# Collect problems found with the password
issues = []
   
# Check if password is too short
if len(password) < 8:    
    issues.append("Too short") 

# Check for at least one upper case letter
if not any(char.isupper() for char in password):
    issues.append("No upper case letter") 

# Check for at least one lower case letter
if not any(char.islower() for char in password):
    issues.append("No lower case letter")

# Check for at least one number 
if not any(char.isdigit() for char in password):
    issues.append("No number")

# Show results 
if not issues:
    print("Password looks ok")
else:
    print("Weak password")
    for issue in issues: 
        print(issue)
